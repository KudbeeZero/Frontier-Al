FRONTIER — Mining Pulse: Expanding Ring on Mine Success
Pass: Globe-MinePulse

Context:
When a mine action succeeds, fire a short-lived visual pulse at that
parcel's location on the 3D globe — a bright ring that expands outward
from the hex and fades to nothing over 600ms. Purely client-side.
No server changes. No polling. Triggered immediately on mine onSuccess.

Architecture:
- GameLayout manages a livePulses state array: { id, lat, lng, startMs }[]
- handleMine and handleMineParcel push a new pulse on onSuccess
- A cleanup effect removes pulses older than 700ms
- PlanetGlobe receives livePulses as an optional prop
- A new MiningPulseLayer component inside the Canvas renders the rings
- Each ring expands from scale 1→4 and fades opacity 1→0 over 600ms
- Multiple simultaneous pulses (mass mine) all render independently

shared/schema.ts, GlobeEventOverlays, BattleArcs, CameraController,

// ── MiningPulseLayer ──────────────────────────────────────────────────────────
// Renders short-lived expanding ring pulses at mined parcel locations.
// Each pulse lives for PULSE_DURATION ms then is removed by GameLayout.

export interface LivePulse {
  id: string;
  lat: number;
  lng: number;
  startMs: number;
}

const PULSE_DURATION = 600; // ms — must match GameLayout cleanup threshold

interface SinglePulseProps {
  pulse: LivePulse;
}

function SinglePulse({ pulse }: SinglePulseProps) {
  const ringRef  = useRef<THREE.Mesh>(null!);
  const glowRef  = useRef<THREE.Mesh>(null!);

  const pos    = useMemo(() => latLngToVec3(pulse.lat, pulse.lng, GLOBE_RADIUS * 1.005), [pulse.lat, pulse.lng]);
  const lookAt = useMemo(() => pos.clone().multiplyScalar(2), [pos]);

  useFrame(() => {
    const elapsed = Date.now() - pulse.startMs;
    const t = Math.min(elapsed / PULSE_DURATION, 1); // 0 → 1

    // Expand: scale 1 → 5
    const scale = 1 + t * 4;
    // Fade: opacity 1 → 0, with a fast initial flash
    const opacity = Math.pow(1 - t, 1.5);

    if (ringRef.current) {
      ringRef.current.scale.setScalar(scale);
      (ringRef.current.material as THREE.MeshBasicMaterial).opacity = opacity * 0.9;
    }
    if (glowRef.current) {
      // Inner glow expands slower and fades faster
      glowRef.current.scale.setScalar(1 + t * 1.8);
      (glowRef.current.material as THREE.MeshBasicMaterial).opacity = opacity * 0.5;
    }
  });

  return (
    <group position={pos}>
      {/* Outer expanding ring */}
      <mesh ref={ringRef} onUpdate={self => self.lookAt(lookAt)}>
        <ringGeometry args={[0.018, 0.030, 32]} />
        <meshBasicMaterial
          color="#00ff6a"
          transparent
          opacity={0.9}
          depthWrite={false}
          side={THREE.DoubleSide}
        />
      </mesh>

      {/* Inner solid glow flash */}
      <mesh ref={glowRef} onUpdate={self => self.lookAt(lookAt)}>
        <circleGeometry args={[0.018, 24]} />
        <meshBasicMaterial
          color="#80ffb0"
          transparent
          opacity={0.5}
          depthWrite={false}
          side={THREE.DoubleSide}
        />
      </mesh>
    </group>
  );
}

interface MiningPulseLayerProps {
  pulses: LivePulse[];
}

function MiningPulseLayer({ pulses }: MiningPulseLayerProps) {
  if (pulses.length === 0) return null;
  return (
    <>
      {pulses.map(pulse => (
        <SinglePulse key={pulse.id} pulse={pulse} />
      ))}
    </>
  );
}

    if (livePulses.length === 0) return;
    const timer = setTimeout(() => {
      const now = Date.now();
      setLivePulses(prev => prev.filter(p => now - p.startMs < 700));
    }, 700);
    return () => clearTimeout(timer);
  }, [livePulses]);
