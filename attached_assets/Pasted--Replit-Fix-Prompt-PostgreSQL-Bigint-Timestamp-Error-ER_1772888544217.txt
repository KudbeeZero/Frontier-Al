### STEP 5: Complete Fix Template

**Use this complete function as your template:**

```javascript
/**
 * Trigger orbital check with fixed timestamp handling
 * @param {string} parcelId - The parcel to check
 * @returns {Promise<Object>} - Database result
 */
async function triggerOrbitalCheck(parcelId) {
  const logger = getLogger('orbital');
  
  try {
    // Step 1: Calculate orbital delay (in milliseconds)
    const orbitalDelay = calculateOrbitalDelay();
    console.log(`[DEBUG] Orbital delay: ${orbitalDelay}`);
    
    // Step 2: Calculate impact time
    const impactTimeMs = Date.now() + orbitalDelay;
    console.log(`[DEBUG] Raw impact time: ${impactTimeMs}`);
    
    // ===== CRITICAL FIX: Round to remove decimals =====
    const impactTimeInt = Math.floor(impactTimeMs);
    console.log(`[DEBUG] Rounded impact time: ${impactTimeInt}`);
    
    // Step 3: Validate timestamp is integer
    if (!Number.isInteger(impactTimeInt)) {
      throw new Error(`Impact time is not an integer: ${impactTimeInt}`);
    }
    
    logger.debug(
      `[ORBITAL-DEBUG] triggerOrbitalCheck | IMPACT triggered | ` +
      `type: ATMOSPHERIC_BURST | parcel: ${parcelId}`
    );
    
    // Step 4: Insert into database with rounded timestamp
    const result = await db.query(
      `INSERT INTO orbital_impacts 
       (parcel_id, impact_time, impact_type, created_at) 
       VALUES ($1, $2, $3, $4)
       RETURNING *`,
      [
        parcelId,
        impactTimeInt,           // ✅ Now an integer
        'ATMOSPHERIC_BURST',
        Math.floor(Date.now())   // ✅ Also rounded
      ]
    );
    
    logger.info(`✅ Orbital impact scheduled: ${parcelId}`);
    return result.rows[0];
    
  } catch (error) {
    logger.error('🔴 Background orbital check error:', {
      message: error.message,
      code: error.code,
      parcelId: parcelId,
      details: error
    });
    throw error;
  }
}
```
