PART 4 — client/src/hooks/useWorldEvents.ts

Add a new exported hook for live events with faster polling:

export function useLiveWorldEvents() {
  return useQuery<WorldEvent[]>({
    queryKey: ["/api/world/events/live"],
    queryFn: async () => {
      const res = await fetch("/api/world/events/recent");
      if (!res.ok) return [];
      return res.json();
    },
    refetchInterval: 5_000,
    staleTime: 3_000,
  });
}
